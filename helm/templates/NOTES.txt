1. Get the application URL:
{{- if .Values.ingress.enabled }}
  Access the app via Ingress at: http://{{ .Values.ingress.host }}
  
  For local testing with port-forward to Ingress controller:
    kubectl port-forward -n ingress-nginx svc/ingress-nginx-controller 80:80
    curl -H "Host: {{ .Values.ingress.host }}" http://localhost/
{{- else if contains "LoadBalancer" .Values.service.type }}
  Get the LoadBalancer IP:
    kubectl get svc --namespace {{ .Release.Namespace }} {{ include "galaxy-bce.fullname" . }}
{{- else if contains "ClusterIP" .Values.service.type }}
  Access via port-forward:
    export POD_NAME=$(kubectl get pods --namespace {{ .Release.Namespace }} -l "app.kubernetes.io/name={{ include "galaxy-bce.name" . }},app.kubernetes.io/instance={{ .Release.Name }}" -o jsonpath="{.items[0].metadata.name}")
    kubectl --namespace {{ .Release.Namespace }} port-forward $POD_NAME 5000:{{ .Values.containerPort }}
    curl http://127.0.0.1:5000/
{{- end }}

2. Check deployment status:
  kubectl get pods -n {{ .Values.namespace.name }}
  kubectl get svc -n {{ .Values.namespace.name }}
  kubectl get ingress -n {{ .Values.namespace.name }}

3. View logs:
  kubectl logs -n {{ .Values.namespace.name }} -l app.kubernetes.io/name={{ include "galaxy-bce.name" . }} -f
